git init  初始化

git status  查看状态

git  add  添加到暂存区

git  commit  -m  添加到get仓库

git  log   查看修改日志

git checkout 文件名   从暂存区恢复文件

git rm --cached  文件名     将文件移除托管

git reset  --hard 文件id    恢复至get仓库存储的id文件时的版本



git分支

git branch   查看分支

git  branch  分支名    创建分支

git checkout    分支名   切换分支

git merge  来源分支名  合并分支
